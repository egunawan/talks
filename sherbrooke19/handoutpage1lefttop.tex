\def\Prufer#1#2#3#4{
	\coordinate (x) at (#1,#2);
	\fill (x) circle (.1);
	\draw[#4] (x) .. controls (#1,#2+2.5) and (#1+.5,#2+3) .. (#1+#3,#2+3);
}

\def\adic#1#2#3#4{
	\coordinate (x) at (#1,#2);
	\fill (x) circle (.1);
	\draw[#4] (x) .. controls (#1,#2+2.5) and (#1-.5,#2+3) .. (#1-#3,#2+3);
}

\def\hadic#1#2#3#4{
	\coordinate (x) at (#1,#2);
	\fill (x) circle (.1);
	\draw[#4] (x) .. controls (#1,#2-2.5) and (#1+.5,#2-3) .. (#1+#3,#2-3);
}

\def\hPrufer#1#2#3#4{

	\fill (x) circle (.1);
	\draw[#4] (x) .. controls (#1,#2-2.5) and (#1-.5,#2-3) .. (#1-#3,#2-3);
}

}
